The Inverted Pyramid

Journalists lead with the essential — who, what, when, where — in the first line. Details follow in decreasing importance.

Editors cut from the bottom. Readers stop whenever they've learned enough. The story still makes sense.

What if code worked the same way?

The Newspaper Style

Headlines

Public methods are short sequences of well-named steps. 5-10 lines. A reader understands the full flow instantly.

Paragraphs

Private methods are the details. Each one has a descriptive name that tells you WHAT happens, not HOW.

Leaves

Complexity lives at the bottom. flatMap chains, fold, try/catch — hidden in private methods. The trunk stays clean.

private Either<RedemptionFailure, Promotion> findPromotion(String code) {
    return promotionRepository
        .findByCodeIgnoreCase(code)
        .map(Either::<RedemptionFailure, Promotion>right)
        .orElse(Either.left(new PromoNotFound(code)));
}

private Either<RedemptionFailure, Promotion> ensureNotRedeemed(
    Promotion promo, String customerId
) {
    return redemptionRepo
        .findByPromoIdAndCustomerId(promo.getId(), toUuid(customerId))
        .map(r -> Either.<RedemptionFailure, Promotion>left(
            new AlreadyRedeemed(customerId)))
        .orElse(Either.right(promo));

export async function POST(req: Request) {
  const body = await req.json();
  if (!body.subscriptionId || !body.paymentMethodId) {
    return Response.json({ error: "Missing fields" }, { status: 400 });
  }
  try {
    const sub = await db.query(
      `SELECT * FROM subscriptions WHERE id = '${body.subscriptionId}'`
    );
    if (!sub) return Response.json({ error: "Not found" }, { status: 404 });
    if (sub.status === "cancelled" || sub.status === "paused") {
      return Response.json({ error: "Inactive" }, { status: 422 });
    }
    const customer = await fetch(
      `${BILLING_API}/customers/${sub.customer_id}`,
      { headers: { Authorization: `Bearer ${BILLING_KEY}` } },
    ).then(r => r.json()) as any;
    const nextRoast = new Date(sub.last_roast_date);
    nextRoast.setDate(nextRoast.getDate() + sub.frequency_weeks * 7);
    const charge = await fetch(`${BILLING_API}/charges`, {
      method: "POST",
      headers: { "Content-Type": "application/json",
        Authorization: `Bearer ${BILLING_KEY}` },
      body: JSON.stringify({
        amount: sub.price_cents, currency: "eur",
        customer: customer.id,
        payment_method: body.paymentMethodId,
        description: `Tostado ${sub.plan_name} - ${sub.bag_size_g}g`,
      }),
    }).then(r => r.json()) as any;
    if (charge.status !== "succeeded") {
      await db.query(`UPDATE subscriptions SET status = 'payment_failed',
        retry_count = retry_count + 1 WHERE id = '${body.subscriptionId}'`);
      await fetch(SLACK_WEBHOOK!, { method: "POST",
        body: JSON.stringify({ text: `Payment failed: ${body.subscriptionId}` })
      });
      return Response.json({ error: charge.failure_reason }, { status: 402 });
    }
    await db.query(`UPDATE subscriptions SET last_roast_date =
      '${nextRoast.toISOString()}', status = 'active' WHERE id = '${body.subscriptionId}'`);
    await redis.del(`customer:${sub.customer_id}:sub`);
    return Response.json({ success: true, nextRoast: nextRoast.toISOString() });
  } catch (err: any) {
    return Response.json({ error: "Internal error" }, { status: 500 });
  }
}

def process_roast_batch(self, batch_id, triggered_by="scheduler"):
    try:
        batch = RoastBatch.objects.select_related("roast_profile", "origin").get(id=batch_id)
        if batch.status not in ("scheduled", "retry"):
            logger.warning(f"Batch {batch_id} unexpected state: {batch.status}")
            return {"skipped": True, "reason": f"status={batch.status}"}
        orders = Order.objects.filter(
            subscription__roast_profile=batch.roast_profile,
            subscription__status="active",
            shipment_batch__isnull=True,
        ).select_related("subscription", "subscription__customer")
        if not orders.exists():
            batch.status = "cancelled"
            batch.cancelled_reason = "no_eligible_orders"
            batch.save()
            return {"cancelled": True}
        total_grams = 0
        for order in orders:
            if order.subscription.bag_size_g:
                total_grams += order.subscription.bag_size_g * order.quantity
            else:
                total_grams += 250 * order.quantity
        inventory = Inventory.objects.get(origin=batch.origin)
        if inventory.available_grams < total_grams:
            batch.status = "blocked"
            batch.blocked_reason = f"Need {total_grams}g, have {inventory.available_grams}g"
            batch.save()
            requests.post(settings.SLACK_WEBHOOK, json={
                "text": f":warning: Batch {batch_id} blocked: {batch.origin.name} "
                        f"({inventory.available_grams}g < {total_grams}g)"
            })
            return {"blocked": True, "deficit": total_grams - inventory.available_grams}
        with transaction.atomic():
            inventory.available_grams -= total_grams
            inventory.reserved_grams += total_grams
            inventory.save()
            batch.status = "roasting"
            batch.total_grams = total_grams
            batch.order_count = orders.count()
            batch.started_at = timezone.now()
            batch.save()
            orders.update(shipment_batch=batch)
        for order in orders:
            send_mail(
                subject=f"Your {batch.roast_profile.name} is being roasted!",
                message=f"Hi {order.subscription.customer.first_name}, your "
                        f"{order.subscription.bag_size_g}g bag is now roasting.",
                from_email=settings.DEFAULT_FROM_EMAIL,
                recipient_list=[order.subscription.customer.email],
            )
        return {"started": True, "orders": orders.count(), "grams": total_grams}
    except Inventory.DoesNotExist:
        raise self.retry(countdown=60)
    except Exception as e:

def process_roast_batch(self, batch_id: str, triggered_by: str = "scheduler"):
    batch = _fetch_batch_if_ready(batch_id)

    orders = _find_eligible_orders(batch)
